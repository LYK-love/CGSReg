# DIAMOND fixed-seed real-ALE eval for TWISTER-replay zero-shot policies.
# Run from $HOME/projects/CGSReg.
# This matches the main-paper local eval protocol:
#   episodes=5, reset_seeds=0,1,2,3,4, deterministic policy.
#
# Set TWEXP_POLICY_ARCHIVE_ROOT to the extracted policy archive root, i.e. the
# directory that contains diamond_twexp/.
